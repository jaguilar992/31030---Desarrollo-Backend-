-- Crear base de datos
CREATE DATABASE mibase;
USE mibase;
-- Crear la tabla
CREATE TABLE usuarios (
    id INT PRIMARY KEY AUTO_INCREMENT NOT NULL,
    nombre VARCHAR(255) NOT NULL ,
    apellido VARCHAR(255) NOT NULL,
    edad INT UNSIGNED NOT NULL,
    email VARCHAR(255) NOT NULL
);
-- insert registros
